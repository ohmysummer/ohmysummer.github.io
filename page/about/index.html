class Ohmycloud {
    has $.location;
    has $.email; 
    has $.qq;   
    has $.title;   
    has @.books;
    
    method gist() {
        qq{
          Location：$!location,
          Email：   $!email,
          QQ：      $!qq,
          Title：   $!title,
        } ~ "  Books：   " ~ @!books.join(',')
    }
}


my $cloudy = Ohmycloud.new(
    location => '北京',
    email    => 'ohmycloudy@gmail.com',
    qq       => '740011611',
    title    => '如果我的勇气是鱼',
    books    => ['暗夜独行客', '一首朋克救地球', '亲爱的阿道夫', '殡葬人的秘密']
);

say $cloudy;
