grammar Scene {
    token TOP { ^ <entry>+ $ } # 一个或多个块
    token entry {
        <head> \s*    # 每个块都有一个标题
        <line>+ \s*   # 每个标题下面有很多非标题行
    }
    token head { '★' \s+ <title> }
    token line {
        [
            <!head>       # 前面不是 head 标题
            .             # 点号匹配换行符
        ]+
     }
    token title { \N+ }
}

my $parsed = Scene.parse($scene);
say  ~$parsed;
